# Execute using:
# tesla_dashcam <Location of video files> --output <Location to store encoded video files> @DIAMOND.txt

# Skip if encoded file already exist.
# --skip_existing

# Delete processed files when video file is encoded.
# --delete_source

# DIAMOND Layout
--layout DIAMOND

# Put clip info bottom left
--halign LEFT --valign BOTTOM

# Output city and timestamp
--text_overlay_fmt "Street: {event_street}\nCity: {event_city}\nDate/Time: {local_timestamp_rolling}"

# Fast forward through portions where there is no motion detected
--motion_only

# Encode with x265
--encoding x265

# Set start offset to 120 seconds
--start_offset 120

# Merge files together
--merge "{layout}_SCALED"

# Merge files together
--merge_timestamp_format "%Y-%m"

# Create map
--title_screen_map

# Scale pillar and repeater cameras to 1/4 normal size
--scale camera=left_pillar 320x240
--scale camera=right_pillar 320x240
--scale camera=left 320x240
--scale camera=right 320x240

# Scale rear camera to 1/2 normal size
--scale camera=rear 640x480
